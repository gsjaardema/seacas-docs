\param{r} = \cmd{AINT} ($x$)
      & truncation: $|x|$ \\
\param{r} = \cmd{ANINT} ($x$)

      & absolute value: $|x|$ \\
\param{r} = \cmd{MOD} ($x$, $y$)

\param{r} = \cmd{DIM} ($x$, $y$)

\param{r} = \cmd{MAX} ($x$, $y$, \ldots)
      & maximum of $x$, $y$, \ldots\ \\
\param{r} = \cmd{MIN} ($x$, $y$, \ldots)
      & minimum of $x$, $y$, \ldots\ \\
\param{r} = \cmd{SQRT} ($x$)

\param{r} = \cmd{EXP} ($x$)
      & exponentiation: e$^{x}$ \\

\param{r} = \cmd{LOG10} ($x$)
      & common logarithm: log$_{10}x$ \\

      & sine $x$ \\
\param{r} = \cmd{COS} ($x$)
      & cosine $x$ \\
\param{r} = \cmd{TAN} ($x$)
      & tangent $x$ \\

      & arc sine $x$ \\

      & arc cosine $x$ \\
\param{r} = \cmd{ATAN} ($x$)
      & arc tangent $x$ \\
\param{r} = \cmd{ATAN2} ($x$, $y$)
      & arc tangent $x/y$ \\
\param{r} = \cmd{SINH} ($x$)
      & hyperbolic sine $x$ \\
\param{r} = \cmd{COSH} ($x$)
      & hyperbolic cosine $x$ \\
\param{r} = \cmd{TANH} ($x$)
      & hyperbolic tangent $x$ \\
