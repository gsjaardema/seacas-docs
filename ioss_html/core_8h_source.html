427 /**
428  An implementation of ``std::basic_string_view`` for pre-C++17. It provides a
429  subset of the API. ``fmt::basic_string_view`` is used for format strings even
430  if ``std::string_view`` is available to prevent issues when a library is
431  compiled with a different ``-std`` option than the client code (which is not
432  recommended).
433  */
434 template <typename Char> class basic_string_view {
435  private:
436  const Char* data_;
437  size_t size_;
438 
439  public:
440  using value_type = Char;
441  using iterator = const Char*;
442 
443  constexpr basic_string_view() FMT_NOEXCEPT : data_(nullptr), size_(0) {}
444 
445  /** Constructs a string reference object from a C string and a size. */
446  constexpr basic_string_view(const Char* s, size_t count) FMT_NOEXCEPT
447  : data_(s),
448  size_(count) {}
449 
450  /**
451  \rst
452  Constructs a string reference object from a C string computing
453  the size with ``std::char_traits<Char>::length``.
454  \endrst
455  */
456  FMT_CONSTEXPR_CHAR_TRAITS
457  FMT_INLINE
458  basic_string_view(const Char* s) : data_(s) {
459  if (detail::const_check(std::is_same<Char, char>::value &&
460  !detail::is_constant_evaluated()))
461  size_ = std::strlen(reinterpret_cast<const char*>(s));
462  else
463  size_ = std::char_traits<Char>::length(s);
464  }
465 
466  /** Constructs a string reference from a ``std::basic_string`` object. */
467  template <typename Traits, typename Alloc>
468  FMT_CONSTEXPR basic_string_view(
469  const std::basic_string<Char, Traits, Alloc>& s) FMT_NOEXCEPT
470  : data_(s.data()),
471  size_(s.size()) {}
472 
473  template <typename S, FMT_ENABLE_IF(std::is_same<
474  S, detail::std_string_view<Char>>::value)>
475  FMT_CONSTEXPR basic_string_view(S s) FMT_NOEXCEPT : data_(s.data()),
476  size_(s.size()) {}
477 
478  /** Returns a pointer to the string data. */
479  constexpr auto data() const -> const Char* { return data_; }
480 
481  /** Returns the string size. */
482  constexpr auto size() const -> size_t { return size_; }
483 
484  constexpr auto begin() const -> iterator { return data_; }
485  constexpr auto end() const -> iterator { return data_ + size_; }
486 
487  constexpr auto operator[](size_t pos) const -> const Char& {
488  return data_[pos];
489  }
490 
491  FMT_CONSTEXPR void remove_prefix(size_t n) {
492  data_ += n;
493  size_ -= n;
494  }
495 
496  // Lexicographically compare this string reference to other.
497  FMT_CONSTEXPR_CHAR_TRAITS auto compare(basic_string_view other) const -> int {
498  size_t str_size = size_ < other.size_ ? size_ : other.size_;
499  int result = std::char_traits<Char>::compare(data_, other.data_, str_size);
500  if (result == 0)
501  result = size_ == other.size_ ? 0 : (size_ < other.size_ ? -1 : 1);
502  return result;
503  }
504 
505  FMT_CONSTEXPR_CHAR_TRAITS friend auto operator==(basic_string_view lhs,
506  basic_string_view rhs)
507  -> bool {
508  return lhs.compare(rhs) == 0;
509  }
510  friend auto operator!=(basic_string_view lhs, basic_string_view rhs) -> bool {
511  return lhs.compare(rhs) != 0;
512  }
513  friend auto operator<(basic_string_view lhs, basic_string_view rhs) -> bool {
514  return lhs.compare(rhs) < 0;
515  }
516  friend auto operator<=(basic_string_view lhs, basic_string_view rhs) -> bool {
517  return lhs.compare(rhs) <= 0;
518  }
519  friend auto operator>(basic_string_view lhs, basic_string_view rhs) -> bool {
520  return lhs.compare(rhs) > 0;
521  }
522  friend auto operator>=(basic_string_view lhs, basic_string_view rhs) -> bool {
523  return lhs.compare(rhs) >= 0;
524  }
525 };

766  private:
767  T* ptr_;
768  size_t size_;

770 
771  protected:

773  FMT_MSC_WARNING(suppress : 26495)

782  buffer(buffer&&) = default;

785  void set(T* buf_data, size_t buf_capacity) FMT_NOEXCEPT {
786  ptr_ = buf_data;
787  capacity_ = buf_capacity;
788  }
789 
790  /** Increases the buffer capacity to hold at least *capacity* elements. */
791  virtual void grow(size_t capacity) = 0;
792 
793  public:
794  using value_type = T;
795  using const_reference = const T&;
796 
797  buffer(const buffer&) = delete;
798  void operator=(const buffer&) = delete;

801  auto end() FMT_NOEXCEPT -> T* { return ptr_ + size_; }
802 
803  auto begin() const FMT_NOEXCEPT -> const T* { return ptr_; }
804  auto end() const FMT_NOEXCEPT -> const T* { return ptr_ + size_; }
805 
806  /** Returns the size of this buffer. */
807  auto size() const FMT_NOEXCEPT -> size_t { return size_; }
808 
809  /** Returns the capacity of this buffer. */
810  auto capacity() const FMT_NOEXCEPT -> size_t { return capacity_; }
811 
812  /** Returns a pointer to the buffer data. */
813  auto data() FMT_NOEXCEPT -> T* { return ptr_; }
814 
815  /** Returns a pointer to the buffer data. */
816  auto data() const FMT_NOEXCEPT -> const T* { return ptr_; }
817 
818  /** Clears this buffer. */
819  void clear() { size_ = 0; }
820 
821  // Tries resizing the buffer to contain *count* elements. If T is a POD type
822  // the new elements may not be initialized.
823  void try_resize(size_t count) {
824  try_reserve(count);
825  size_ = count <= capacity_ ? count : capacity_;
826  }
827 
828  // Tries increasing the buffer capacity to *new_capacity*. It can increase the
829  // capacity by a smaller amount than requested but guarantees there is space
830  // for at least one additional element either by increasing the capacity or by
831  // flushing the buffer if it is full.
832  void try_reserve(size_t new_capacity) {
833  if (new_capacity > capacity_) grow(new_capacity);
834  }
835 
836  void push_back(const T& value) {
837  try_reserve(size_ + 1);
838  ptr_[size_++] = value;
839  }

845  template <typename I> auto operator[](I index) const -> const T& {
846  return ptr_[index];

2415  return handler.on_error("unmatched '}' in format string");
2416  handler.on_text(begin, p);
2417  begin = ++p;
2418  }
2419  }
2420  handler.on_text(begin, end);
2421  return;
2422  }
2423  struct writer {
2424  FMT_CONSTEXPR void operator()(const Char* pbegin, const Char* pend) {
2425  if (pbegin == pend) return;
2426  for (;;) {
2427  const Char* p = nullptr;
2428  if (!find<IS_CONSTEXPR>(pbegin, pend, '}', p))
2429  return handler_.on_text(pbegin, pend);
2430  ++p;
2431  if (p == pend || *p != '}')
2432  return handler_.on_error("unmatched '}' in format string");
2433  handler_.on_text(pbegin, p);
2434  pbegin = p + 1;
2435  }
2436  }
2437  Handler& handler_;
2438  } write{handler};
2439  while (begin != end) {
2440  // Doing two passes with memchr (one for '{' and another for '}') is up to
2441  // 2.5x faster than the naive one-pass implementation on big format strings.
2442  const Char* p = begin;
2443  if (*begin != '{' && !find<IS_CONSTEXPR>(begin + 1, end, '{', p))
2444  return write(begin, end);
2445  write(begin, p);
2446  begin = parse_replacement_field(p, end, handler);
2447  }
2448 }

2782  case detail::type::bool_type:
2783  if (!specs_.type || specs_.type == 's') break;
2784  FMT_FALLTHROUGH;
2785  case detail::type::int_type:
2786  case detail::type::uint_type:

2792  break;

2799  else

2805  else

2807  break;

2811  else
