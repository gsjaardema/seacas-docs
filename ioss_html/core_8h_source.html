354 /**
355  An implementation of ``std::basic_string_view`` for pre-C++17. It provides a
356  subset of the API. ``fmt::basic_string_view`` is used for format strings even
357  if ``std::string_view`` is available to prevent issues when a library is
358  compiled with a different ``-std`` option than the client code (which is not
359  recommended).
360  */
361 template <typename Char> class basic_string_view {
362  private:
363  const Char* data_;
364  size_t size_;
365 
366  public:
367  using value_type = Char;
368  using iterator = const Char*;
369 
370  constexpr basic_string_view() FMT_NOEXCEPT : data_(nullptr), size_(0) {}
371 
372  /** Constructs a string reference object from a C string and a size. */
373  constexpr basic_string_view(const Char* s, size_t count) FMT_NOEXCEPT
374  : data_(s),
375  size_(count) {}
376 
377  /**
378  \rst
379  Constructs a string reference object from a C string computing
380  the size with ``std::char_traits<Char>::length``.
381  \endrst
382  */
383 #if __cplusplus >= 201703L // C++17's char_traits::length() is constexpr.
384  FMT_CONSTEXPR
385 #endif
386  basic_string_view(const Char* s)
387  : data_(s), size_(std::char_traits<Char>::length(s)) {}
388 
389  /** Constructs a string reference from a ``std::basic_string`` object. */
390  template <typename Traits, typename Alloc>
391  FMT_CONSTEXPR basic_string_view(
392  const std::basic_string<Char, Traits, Alloc>& s) FMT_NOEXCEPT
393  : data_(s.data()),
394  size_(s.size()) {}
395 
396  template <typename S, FMT_ENABLE_IF(std::is_same<
397  S, detail::std_string_view<Char>>::value)>
398  FMT_CONSTEXPR basic_string_view(S s) FMT_NOEXCEPT : data_(s.data()),
399  size_(s.size()) {}
400 
401  /** Returns a pointer to the string data. */
402  constexpr const Char* data() const { return data_; }
403 
404  /** Returns the string size. */
405  constexpr size_t size() const { return size_; }
406 
407  constexpr iterator begin() const { return data_; }
408  constexpr iterator end() const { return data_ + size_; }
409 
410  constexpr const Char& operator[](size_t pos) const { return data_[pos]; }
411 
412  FMT_CONSTEXPR void remove_prefix(size_t n) {
413  data_ += n;
414  size_ -= n;
415  }
416 
417  // Lexicographically compare this string reference to other.
418  int compare(basic_string_view other) const {
419  size_t str_size = size_ < other.size_ ? size_ : other.size_;
420  int result = std::char_traits<Char>::compare(data_, other.data_, str_size);
421  if (result == 0)
422  result = size_ == other.size_ ? 0 : (size_ < other.size_ ? -1 : 1);
423  return result;
424  }
425 
426  friend bool operator==(basic_string_view lhs, basic_string_view rhs) {
427  return lhs.compare(rhs) == 0;
428  }
429  friend bool operator!=(basic_string_view lhs, basic_string_view rhs) {
430  return lhs.compare(rhs) != 0;
431  }
432  friend bool operator<(basic_string_view lhs, basic_string_view rhs) {
433  return lhs.compare(rhs) < 0;
434  }
435  friend bool operator<=(basic_string_view lhs, basic_string_view rhs) {
436  return lhs.compare(rhs) <= 0;
437  }
438  friend bool operator>(basic_string_view lhs, basic_string_view rhs) {
439  return lhs.compare(rhs) > 0;
440  }
441  friend bool operator>=(basic_string_view lhs, basic_string_view rhs) {
442  return lhs.compare(rhs) >= 0;
443  }
444 };

576  int next_arg_id = 0)
577  : ErrorHandler(eh), format_str_(format_str), next_arg_id_(next_arg_id) {}

580  Returns an iterator to the beginning of the format string range being
581  parsed.
582  */

584 
585  /**
586  Returns an iterator past the end of the format string range being parsed.
587  */
588  constexpr iterator end() const FMT_NOEXCEPT { return format_str_.end(); }
589 
590  /** Advances the begin iterator to ``it``. */
591  FMT_CONSTEXPR void advance_to(iterator it) {
592  format_str_.remove_prefix(detail::to_unsigned(it - begin()));
593  }

595  /**
596  Reports an error if using the manual argument indexing; otherwise returns
597  the next argument index and switches to the automatic indexing.
598  */
599  FMT_CONSTEXPR int next_arg_id() {
600  // Don't check if the argument id is valid to avoid overhead and because it
601  // will be checked during formatting anyway.
602  if (next_arg_id_ >= 0) return next_arg_id_++;
603  on_error("cannot switch from manual to automatic argument indexing");
604  return 0;
605  }
606 
607  /**
608  Reports an error if using the automatic argument indexing; otherwise
609  switches to the manual indexing.
610  */
611  FMT_CONSTEXPR void check_arg_id(int) {
612  if (next_arg_id_ > 0)
613  on_error("cannot switch from automatic to manual argument indexing");
614  else
615  next_arg_id_ = -1;
616  }
617 
618  FMT_CONSTEXPR void check_arg_id(basic_string_view<Char>) {}
619 
620  FMT_CONSTEXPR void on_error(const char* message) {
621  ErrorHandler::on_error(message);
622  }

624  constexpr ErrorHandler error_handler() const { return *this; }

671 template <typename T> class buffer {
672  private:
673  T* ptr_;
674  size_t size_;
675  size_t capacity_;
676 
677  protected:
678  // Don't initialize ptr_ since it is not accessed to save a few cycles.
679  FMT_SUPPRESS_MSC_WARNING(26495)
680  buffer(size_t sz) FMT_NOEXCEPT : size_(sz), capacity_(sz) {}
681 
682  buffer(T* p = nullptr, size_t sz = 0, size_t cap = 0) FMT_NOEXCEPT
683  : ptr_(p),
684  size_(sz),
685  capacity_(cap) {}
686 
687  ~buffer() = default;
688 
689  /** Sets the buffer data and capacity. */
690  void set(T* buf_data, size_t buf_capacity) FMT_NOEXCEPT {
691  ptr_ = buf_data;
692  capacity_ = buf_capacity;
693  }
694 
695  /** Increases the buffer capacity to hold at least *capacity* elements. */
696  virtual void grow(size_t capacity) = 0;
697 
698  public:
699  using value_type = T;
700  using const_reference = const T&;
701 
702  buffer(const buffer&) = delete;
703  void operator=(const buffer&) = delete;
704 
705  T* begin() FMT_NOEXCEPT { return ptr_; }
706  T* end() FMT_NOEXCEPT { return ptr_ + size_; }
707 
708  const T* begin() const FMT_NOEXCEPT { return ptr_; }
709  const T* end() const FMT_NOEXCEPT { return ptr_ + size_; }
710 
711  /** Returns the size of this buffer. */
712  size_t size() const FMT_NOEXCEPT { return size_; }
713 
714  /** Returns the capacity of this buffer. */
715  size_t capacity() const FMT_NOEXCEPT { return capacity_; }
716 
717  /** Returns a pointer to the buffer data. */
718  T* data() FMT_NOEXCEPT { return ptr_; }
719 
720  /** Returns a pointer to the buffer data. */
721  const T* data() const FMT_NOEXCEPT { return ptr_; }
722 
723  /** Clears this buffer. */
724  void clear() { size_ = 0; }
725 
726  // Tries resizing the buffer to contain *count* elements. If T is a POD type
727  // the new elements may not be initialized.
728  void try_resize(size_t count) {
729  try_reserve(count);
730  size_ = count <= capacity_ ? count : capacity_;
731  }
732 
733  // Tries increasing the buffer capacity to *new_capacity*. It can increase the
734  // capacity by a smaller amount than requested but guarantees there is space
735  // for at least one additional element either by increasing the capacity or by
736  // flushing the buffer if it is full.
737  void try_reserve(size_t new_capacity) {
738  if (new_capacity > capacity_) grow(new_capacity);
739  }
740 
741  void push_back(const T& value) {
742  try_reserve(size_ + 1);
743  ptr_[size_++] = value;
744  }

746  /** Appends data to the end of the buffer. */
747  template <typename U> void append(const U* begin, const U* end);
748 
749  template <typename I> T& operator[](I index) { return ptr_[index]; }
750  template <typename I> const T& operator[](I index) const {
751  return ptr_[index];
752  }
